public class ThrowableCaptor {
  public static <T> Throwable captureThrowable(Consumer<T> consumer, T el){
    Throwable result = null;
    try {
      consumer.accept(el);
    }catch(Exception e) {
      result = e;
    }
    return result;
  }
}

                                .resolve("org.hibernate:hibernate-entitymanager:4.2.15.Final")
